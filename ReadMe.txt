CREATE PROCEDURE [dbo].[Add_Factura]    
	 @ID_Institucion NVARCHAR(25),
	 @ID_Cliente NVARCHAR(25),
	 @ID_Vehiculo NVARCHAR(25),
	 @Nombre_Cliente NVARCHAR(250),

	 @Monto_pagar NVARCHAR(250),

    BEGIN

		SET @Id = SCOPE_IDENTITY();   
        SELECT  @Id AS ReservaID;    
    END;
